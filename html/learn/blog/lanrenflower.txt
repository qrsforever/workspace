npm install hexo-generator-json-content --save

# 域名CNAME 自动生成 或者是将CNAME放到source目录下(推荐)

ln -s hexo-blog/source source
ln -s hexo-blog/themes/hexo-theme-indigo themes/hexo-theme-indigo
